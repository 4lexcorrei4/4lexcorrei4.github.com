type symbol = char;;       (* our symbols are represented by chars *)
type word = symbol list;;  (* our words are represented by lists of symbols *)

type state = string;;      (* our states are represented by strings *)
type states = state list;;

type transition =
    state   (* state *)
  * symbol  (* consumed input symbol *)
  * state   (* next state *)
;;
type transitions = transition list;;

type fAutomaton = {
    initialState: state;       (* Initial state *)
    transitions: transitions;  (* Transition relation *)
    acceptStates: states       (* Accept states *)
};;

let rec uniq l =
	match l with
	| [] -> []
	| [x] -> [x]
	| x::y::xs -> (if x = y then [] else [x]) @ uniq (y::xs)
;;

let canonical l =
    uniq (List.sort compare l)
;;

let addAll symb =
    List.map (fun l -> symb::l)
;;

let flatMap f l =
    List.flatten (List.map f l)
;;


(* PUBLIC FUNCTIONS *)

let getStatesCheck tuple =
	match tuple with
	| (s1, _, s2) -> s1 :: s2 :: []
;;

let getStates fa =
	canonical ((flatMap getStatesCheck fa.transitions) 
		@ (fa.initialState :: fa.acceptStates))
;;

let getAlphabet fa =
  canonical (List.map (fun (_, a, _) -> a) fa.transitions)
;;

let gcut s ts =
  List.partition (fun (s1, _, _) -> s1 = s) ts
;;

let transitionsFromStates fa =
	canonical(List.map (fun (s1, a, _) -> (s1, a)) fa.transitions)
;;

let determinism fa =
	compare (List.length fa.transitions)
		(List.length (transitionsFromStates fa)) = 0
;;

let rec reachableRec s ts =
	let (t1, t2) = gcut s ts in
		s::reachableNextStates t1 t2

and reachableNextStates t1 t2 =
	match t1 with
	| [] -> []
	| (_, _, s2)::t1t -> reachableRec s2 t2 @ reachableNextStates t1t t2
;;

let reachable fa =
	canonical (reachableRec fa.initialState fa.transitions) 
;;

let rec hasElem l ast = 
	match l with
	| [] -> false
	| x::xs -> if List.mem x ast then true
			else hasElem xs ast
;;

let rec productiveRec st ts ast =
  	match st with
	| [] -> []
	| x::xs -> if hasElem (reachableRec x ts) ast
				then x::productiveRec xs ts ast
				else productiveRec xs ts ast
;;

let productive fa =
  canonical (productiveRec (getStates fa) fa.transitions fa.acceptStates)
;;

let acceptSearch s1 a tuple =
	match tuple with
	| (ss, sa, _) -> s1 = ss && a = sa
;;

let rec acceptProccess w s ass ts =
	match w with
	| [] -> List.mem s ass
	| h::t -> if List.exists (acceptSearch s h) ts then
				let (_, _, s2) = List.find (acceptSearch s h) ts in
					acceptProccess t s2 ass ts
			else false
;;

let accept w fa =
	acceptProccess w fa.initialState fa.acceptStates fa.transitions
;;

let accept2Check s1 a tuple =
	match tuple with
	| (ss, sa, _) -> s1 = ss && a = sa
;;

let rec accept2Proccess w s ass ts =
	match w with
	| [] -> List.mem s ass
	| hw::tw -> if List.exists (accept2Check s hw) ts then
				accept2Partition hw tw s ass ts ts
				else false
and accept2Partition t w s ass ts ats =
	match ts with
	| [] -> false
	| (is, a, fs)::xs -> if accept2Check s t (is, a, fs) 
	then accept2Proccess w fs ass ats || accept2Partition t w s ass xs ats
						else accept2Partition t w s ass xs ats
;;

let accept2 w fa =
  accept2Proccess w fa.initialState fa.acceptStates fa.transitions
;;


let rec add alph alphs n =
	if n=0 then [[]]
	else match alphs with
		| [] -> []
		| x::xs -> (addAll x (add alph alph (n-1))) @ (add alph xs n)
;;

let rec generateProccess l fa =
	match l with
	| [] -> []
	| x::xs -> if accept2 x fa then
				x::(generateProccess xs fa)
				else generateProccess xs fa
;;

let generate n fa =
  canonical (generateProccess (add (getAlphabet fa) (getAlphabet fa) n) fa)
;;
